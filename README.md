CREATE TABLE TBL_SESSAO(
    ID_SESSAO BIGINT PRIMARY KEY IDENTITY,
    DT_DATA DATE NOT NULL,
    NR_HORARIO TIME NOT NULL,
    ID_FILME BIGINT FOREIGN KEY(ID_FILME) REFERENCES TBL_FILME(ID_FILME),
    NR_PRECO FLOAT NOT NULL,
    TX_SALA VARCHAR(300) NOT NULL
);
